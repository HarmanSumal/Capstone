import matplotlib.pyplot as plt
import numpy as np

# Define the classes and their corresponding indices
classes = ['Excellent', 'Good', 'Poor', 'Very Poor', 'Non-drinkable']
indices = [0, 1, 2, 3, 4]

# Provide precision, recall, and F1-score values for each class
precision = [0.00, 0.84, 0.64, 0.00, 0.95]
recall = [0.00, 0.80, 0.92, 0.00, 0.96]
f1_score = [0.00, 0.82, 0.75, 0.00, 0.96]


# Plot the horizontal bar chart without boundaries
bar_width = 0.25
index = np.arange(len(classes))

plt.barh(index, precision, bar_width, label='Precision', color='r', edgecolor='none')
plt.barh(index + bar_width, recall, bar_width, label='Recall', color='g', edgecolor='none')
plt.barh(index + 2*bar_width, f1_score, bar_width, label='F1-score', color='b', edgecolor='none')

# Annotate the metrics in the center of each bar with percentages
for i in index:
    plt.annotate(f"{precision[i]*100:.2f}%", xy=(precision[i], i), xytext=(3, 0),
                 textcoords='offset points', ha='left', va='center', color='black')
    plt.annotate(f"{recall[i]*100:.2f}%", xy=(recall[i], i + bar_width), xytext=(3, 0),
                 textcoords='offset points', ha='left', va='center', color='black')
    plt.annotate(f"{f1_score[i]*100:.2f}%", xy=(f1_score[i], i + 2*bar_width), xytext=(3, 0),
                 textcoords='offset points', ha='left', va='center', color='black')

plt.yticks(index + bar_width, classes)
plt.title('Precision, Recall, and F1-score for each class With Logistic Regression')
plt.legend(loc='center left', bbox_to_anchor=(1, 0.10))

# Remove borders and ticks
plt.gca().spines['top'].set_visible(False)
plt.gca().spines['right'].set_visible(False)
plt.gca().spines['bottom'].set_visible(False)
plt.gca().spines['left'].set_visible(False)
plt.tick_params(axis='both', which='both', length=0)
plt.xticks([])  # Remove x-axis tick labels

plt.tight_layout()
plt.show()

In the above bar graph, based on the high precision, recall and f1 score our logistic regression model is predicting non-drinkable class.
